def showTasks():
    pos = 0
    while pos == 0:
        pos = ShowLists()
    TODO = TaskList.lists[pos].items
    Choice = TaskList.lists[pos]
    if TODO:
        print("Your exercises:")
        pos = 1
        for i in TODO:
            print(f"{pos}.  {i}")
            pos += 1
    else:
            print("Congratulations! You have nothing to do")
    return Choice

def ShowLists():
    print("Please choose your task list of choice.")
    pos = 1 #keeping track of our position
    for i in TaskList.lists[1:]: #the 1: will only go from the position 1 and on, this is so users don't get confused heh
        print(f"{pos}. {i.name}") #requires python 3.6+, can be replaced with %s or multiple print statements.
        pos += 1
    print("0.  Create new task list")
    pos = int(input("Your selection? "))
    if pos == 0:
        lname = input("Enter your task name of choice!: ")
        TaskList(lname)
    return pos

class TaskList:
    def __init__(self, name):
        cls = type(self)
        cls.lists.append(self)
        self.items = []
        self.name = name
    def CreateTask(self, name):
        self.items.append(name)
    def DeleteTask(self, pos):
        del self.items[pos]
    lists = [None]

def checkTask(item):
    TODO = item.items
    print("Which task do you want to select?")
    number = int(input("Number: "))
    print("This task will be removed. Is this OK?", end=" ")
    print("\n" + TODO[number - 1], end=" ")
    really = input("Y/N: ")
    if really == "Y" or really == "y":
        item.DeleteTask(number - 1)
    else:
        print("Canceled.")
