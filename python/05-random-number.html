#!/usr/bin/python3
# 05-random-number.py

from random import randint

number = randint(1, 100)

print("I've chosen a number from 1 to 100")

number_input = ""

while number_input != number:
    number_input = int(input("Number: "))
    if number_input < number:
        print("Number is too low")
    elif number_input > number:
        print("Number is too high")
    else:
        print("You've found the right number! It's " + str(number) + ".")
