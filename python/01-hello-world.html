#!/usr/bin/python3
# 01-hello-world.py

from time import *

name = input("name: ")
year = int(input("year of birth: "))

age = gmtime().tm_year - year

sentence = "Hello " + name + ". You are " + str(age) + " years old."

print(sentence)
