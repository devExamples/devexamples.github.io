def start():
    global a_b
    showTasks()
    print("What do you want to do?")
    print("(A) Create Task")
    print("(B) Check Task")
    a_b = input("A or B: ")
    if a_b == "a" or a_b == "A":
        createTask()
    elif a_b == "b" or a_b == "B":
        checkTask()
    else:

def showTasks():
    TODO = Task.TODO
    if TODO:
        print("Your exercises:")
        for i in TODO:
            print("- ", end=" ")
            print(i)
    else:
            print("Congratulations! You have nothing to do")

def createTask():
    print("Which task do you want to add?")
    task = input("Task: ")
    Task.CreateTask(task)

class Task:
    @classmethod
    def CreateTask(cls, name):
        cls.TODO.append(name)
    TODO = []

def checkTask():
    TODO = Task.TODO
    print("Which task do you want to select?")
    number = int(input("Number: "))
    print("This task will be removed. Is this OK?", end=" ")
    print(TODO[number - 1], end=" ")
    really = input("Y/N: ")
    if really == "Y" or really == "y":
        del TODO[number - 1]
