#!/usr/bin/python3
# rock-paper-scissors.py

from random import randint

points = [0, 0]
figures = ["Rock", "Paper", "Scissors"]
play = True

def check(a, b):
    a = a.lower() #make variable `a' lowercase
    b = b.lower() #make variable `b' lowercase
    if a == "scissors": #if variable `a' is equal to "scissors"
        if b == "scissors":
            result("ab") #run function result with parameter "ab"
        elif b == "rock":
            result("b")
        elif b == "paper":
            result("a")
    elif a == "rock":
        if b == "scissors":
            result("a")
        elif b == "rock":
            result("ab")
        elif b == "paper":
            result("b")
    elif a == "paper":
        if b == "scissors":
            result("b")
        elif b == "rock":
            result("a")
        elif b == "paper":
              result("ab")

def result(winner): #define function
    global points #make variable `points' able to be used outside of function result()
    if winner == "a":
        points[0] += 1 #add 1 to first element of list `points'
    elif winner == "ab":
        points[0] += 1
        points[1] += 1
    else:
        points[1] += 1
    print("You: " + str(points[0]) + " & Computer: " + str(points[1]) + ".")

def finish():
    global play
    if points[0] > points[1]:
        print("Congratulations! You're the winner.")
    elif points[0] == points[1]:
        print("Tie! Both of you won.")
    else:
        print("Damage! Unfortunately the computer won.")
    print("Again?")
    decision = 0
    while decision not in ["Yes", "No", "yes", "no"]:
        decision = input("Yes / No: ")
    if decision.lower() == "yes":
        play = True
    else:
        play = False

while play:
    print("* Rock-Paper-Scissors *")

    for runde in range(3):
        print("Round " + str(runde + 1) + ":")
        player = 0
        while player not in ["rock", "scissors",  "paper"]:
            player = input("Your choice: ").lower() #ask for a choice and make it lowercase
        computer = figures[randint(0, 2)]
        print("Computer: " + computer)
        check(player, computer)

    finish()
