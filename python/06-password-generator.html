#!/usr/bin/python3
# 06-passwort-generator.py

from random import *
from string import *

go = True

def generatePassword(length):
    global go
    password = ""
    symbols = ["!", "&", "/", "(", ")", "=", "?", ",", ";", ".", ":", "-", "_"]

    for i in range(length):
        type = choice(["letter", "symbol", "number"])
            if type == "letter":
                password += choice(ascii_letters)
            elif type == "symbol":
                password += choice(symbols)
            else:
                password += choice(digits)

        return password
